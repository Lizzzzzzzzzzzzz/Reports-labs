def textFile():
    count = 0
    l = int(input("Введіть довжину"))
    with open('file.txt', "r", encoding="UTF-8") as file:
        s = file.read()
        k = s.split()
        for el in k:
            if len(el) < l:
                k.remove(el)
                count += 1
    #s = s.replace(str(k), "")
    with open('file.txt', "w", encoding="UTF-8") as f:
        f.write(s)

textFile()
